    node = TreeNode()

    root = node

    for i in bookAry[1:]:
        node = TreeNode()
        node.data= i[0]
        current = root
        while True:
            if current.data &gt; i[0]:
                if current.left ==None:
                    current.left = node
                    break
                current = current.left
            
            elif current.data &lt; i[0]:
                if current.right ==None:
                    current.right = node
                    break
								 current = current.right

    return root

<h2>2. 이진 탐색 트리 데이터 검색</h2>
<pre class="nsis"><code>def find_node(root, findName):
    current = root
    
    while True:
        if current.data == findName:
            result = findName + '을(를) 찾음.'
            break
        elif current.data &gt; findName:
            if current.left == None:
                result = findName + '이(가) 트리에 없음'
                break
            current = current.left
        else:
            if current.right == None:
                result = findName + '이(가) 트리에 없음'
                break
            current = current.right
    return result
</code></pre>
<h2>3. 전위 순회</h2>
<pre class="crmsh"><code>def preorder(node):
    if node ==None:
        return 
    print(node.data, end = ' ')
    preorder(node.left)
    preorder(node.right)
</code></pre>
<h2>4. 중위 순회</h2>

<h2>5. 후위 순회</h2>

<h2>6. root 받아 left, right 반전 후 root 반환 함수</h2>
<pre class="xquery"><code>def invertTree(root):
    if root==None:
        return

    root.right, root.left = root.left, root.right

    invertTree(root.left)
    invertTree(root.right)

    return root
</code></pre>
<h2>7. 큰 값부터 작은 값으로 순회, 배열 누적 함수</h2>
<pre class="reasonml"><code>def traverse_node(root, result):
    if root == None:
        return
    traverse_node(root.right, result)
    result.append(root.data)
    traverse_node(root.left, result)

    return result
</code></pre>
<h2>8. 이진 탐색 트리의 원리</h2>
<ul>
<li>연결리스트의 <b>head노드</b>가 이진 트리에서는 <b>root노드</b>로 표현된다.</li>
<li>root노드의 data보다 <b>작은 값</b>이 들어오면 <b>left</b>, <b>큰 값</b>이 들어오면 <b>right</b> 이동</li>
<li>왼쪽 서브 트리와 오른쪽 서브 트리도 똑같이 구성</li>
<li>모든 노드 값은 <b>중복</b>되지 않는다.</li>
</ul>
